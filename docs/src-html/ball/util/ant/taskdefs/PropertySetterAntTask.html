021import org.apache.tools.ant.BuildException;
022import org.apache.tools.ant.Task;
023
024/**
025 * Interface to provide common default methods for {@link Task}s that may
026 * assign property values.
027 *
028 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
029 */
030public interface PropertySetterAntTask extends AntTaskMixIn {
031    String getProperty();
032    void setProperty(String property);
033
034    /**
035     * Method to get the value to assign to the property.
036     *
037     * @return  The property value.
038     *
039     * @throws  Exception       As specified by implementing subclass.
040     */
041    Object getPropertyValue() throws Exception;
042
043    default void execute() throws BuildException {
044        try {
045            Task task = (Task) this;
046            String key = getProperty();
047            Object value = getPropertyValue();
048
049            if (key != null) {
050                if (value != null) {
051                    task.getProject().setNewProperty(key, value.toString());
052                }
053            } else {
054                task.log(String.valueOf(value));
055            }
056        } catch (BuildException exception) {
057            throw exception;
058        } catch (RuntimeException exception) {
059            throw exception;
060        } catch (Exception exception) {
061            throw new BuildException(exception);
062        }
063    }
064}
